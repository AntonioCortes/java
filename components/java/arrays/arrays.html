foreach y for para recorrer arrays: 

String[] nombresList = new String[]{"Juan","Juan","Juan"};

for(String nombre : nombresList){
    System.out.println(nombre);
}

for(int i=0; i<nombresList.size(); i++){
    System.out.println(
        nombreList[i];
        );
}
